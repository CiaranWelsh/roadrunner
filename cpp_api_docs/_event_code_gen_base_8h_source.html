63 template <typename Derived>
64 llvm::Value *EventCodeGenBase<Derived>::codeGen()
65 {
66  // make the set init value function
67  llvm::Type *argTypes[] = {
68  llvm::PointerType::get(ModelDataIRBuilder::getStructType(this->module), 0),
69  llvm::Type::getInt32Ty(this->context),
70  llvm::Type::getDoublePtrTy(this->context)
71  };
72 
73  const char *argNames[] = {
74  "modelData", "eventIndx", "data"
75  };
76 
77  llvm::Value *args[] = {0, 0, 0};
78 
79  llvm::Type *retType = llvm::Type::getVoidTy(context);
80 
81  llvm::BasicBlock *entry = this->codeGenHeader(Derived::FunctionName, retType,
82  argTypes, argNames, args);
83 
84  const libsbml::ListOfEvents *events = this->model->getListOfEvents();
85 
86  // default, return NaN
87  llvm::BasicBlock *def = llvm::BasicBlock::Create(this->context, "default", this->function);
88  this->builder.SetInsertPoint(def);
89 
90  this->builder.CreateRetVoid();
91 
92  // write the switch at the func entry point, the switch is also the
93  // entry block terminator
94  this->builder.SetInsertPoint(entry);
95 
96  llvm::SwitchInst *s = this->builder.CreateSwitch(args[1], def, events->size());
97 
98  for (uint i = 0; i < events->size(); ++i)
99  {
100  char block_name[64];
101  std::sprintf(block_name, "event_%i_block", i);
102  llvm::BasicBlock *block = llvm::BasicBlock::Create(this->context, block_name, this->function);
103  this->builder.SetInsertPoint(block);
104 
105  const libsbml::Event *event = events->get(i);
106 
107  bool cont = static_cast<Derived*>(this)->eventCodeGen(args[0], args[2], event);
108 
109  this->builder.CreateRetVoid();
110  s->addCase(llvm::ConstantInt::get(llvm::Type::getInt32Ty(this->context), i), block);
111 
112  if (!cont) break;
113  }
114 
115  return this->verifyFunction();
116 }
