36  /**
37  * Set the value for an arbitrary key.
38  *
39  * The Variant object has a large number of creation operators so
40  * you only have to give the value as native C++ object i.e.
41  *
42  * @code
43  * Dictionary*d = (...);
44  * d->setItem("someIntKey", 1);
45  * d->setItem("someStrKey", string("some string"));
46  * d->setItem("someFloatVal, 0.123);
47  * @endcode
48  */
49  virtual void setItem(const std::string& key, const rr::Variant& value) = 0;

51  /**
52  * Get a value. Variants are POD. If the key does not exist, an std::exception
53  * will be thrown.
54  *
55  * The Variant object has the assigment operator overloaded so
56  * it will automatically convert to any native C++ object. If the
57  * Variant can not be converted, it will raise an exception, i.e.
58  *
59  * @code
60  * Dictionary* d = ...;
61  * int a = d->getItem("someIntKey");
62  * string s = d->getItem("someStrVal");
63  * double val = d->getItem("someDoubleVal");
64  * @endcode
65  */
