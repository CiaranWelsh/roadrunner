186  /**
187  * A set of flags that are common to all integrators. The SimulateOptions::integratorFlags
188  * field is a bitmask that contains these values.
189  */
190  enum IntegratorFlags
191  {
192  /**
193  * Is the model a stiff system? setting this to stiff causes
194  * RoadRunner to load a stiff solver which could potentially be
195  * extremly slow
196  */
197  STIFF = (0x1 << 0), // => 0x00000001

199  /**
200  * The MULTI_STEP option tells the solver to take a series of internal steps
201  * and then return the solution at the point reached by that step.
202  *
203  * In simulate, this option will likely be slower than normal mode,
204  * but may be useful to monitor solutions as they are integrated.
205  *
206  * This is intended to be used in combination with the
207  * IntegratorListener. It this option is set, and there is a
208  * IntegratorListener set, RoadRunner::integrate will run the
209  * integrator in a series of internal steps, and the listner
210  * will by notified at each step.
211  *
212  * Highly Experimental!!!
213  */
214  MULTI_STEP = (0x1 << 1), // => 0x00000010

216  /**

218  * integrator to best choose an adaptive time step and the resulting
219  * matrix will have a non-uniform time column

242 /**
243  * A factory class to create an integrator.
244  *
245  * This class also has a number of static method which return the list of
246  * available integrators, and methods to return dictionaries of the available
247  * options each Integrator supports.
248  *
249  * The Dictionary references that are returned are borrowed references, if one
250  * wants to modify them, a new SimululateOptions or DictionaryImpl may be created
251  * using their copy constructors.
252  *
253  * When a new mutable dictionary is created, it may be used as the
254  * argument to RoadRunner::simulate();
255  *
256  * The integrator dictionary contain a set of descriptor keys / values, these
257  * are the standard key names, but have a ".description" or ".hint" suffix.
258  *
259  * For example, to iterate through all of the available integrator options,
260  * and display them to the console, one would:
261  * @code
262  * // get a list of all of the integrator options
263  * vector<const Dictionary*> opts =
264  * IntegratorFactory::getIntegratorOptions();
265  *
266  * // iterate through the integrator options
267  * for(int i = 0; i < opts.size(); ++i) {
268  * // each dictionary will contain all the keys that
269  * // are valid for a particular integrator.
270  * const Dictionary &d = *opts[i];
271  * vector<string> keys = d.getKeys();
272  * for(int j = 0; j < keys.size(); ++j) {
273  * string key = keys[j];
274  * string item = d.getItem(key);
275  * cout << "key: " << key << ", value: " << item << std::endl;
276  * }
277  * }
278  * @endcode
279  *
280  * Notice here, the line, `string item = d.getItem(key)`, this implicitly calls
281  * the conversion operator to automatically convert the contents of the Variant to
282  * a string. Any type that is stored in a Variant can be converted to a string.
283  *
284  * Each Dictionary object will contain a "integrator" key, this identifies the
285  * integrator. It will also contain a "integrator.description" and an "integrator.hint"
286  * keys, these provide a full description and a hint for that particular integrator.
287  *
288  * One may also get a list of all the available integrators via the
289  * getIntegratorNames() method, and use this integrator name to get the
290  * list of options available for a particular integrator.
291  *
292  * The values in each integrator options dictionary are those of the current
293  * default values. The default values may be modified via the Config class.
294  */
