31 class RR_DECLSPEC LoadSBMLOptions : public BasicDictionary
32 {
33 public:
34  enum ModelGeneratorOpt
35  {
36  /**
37  * perform conservation analysis.
38  *
39  * This causes a re-ordering of the species, so results generated
40  * with this flag enabled can not be compared index wise to results
41  * generated otherwise.
42  *
43  * currently only implemented with the C code generating model
44  */
45  CONSERVED_MOIETIES = (0x1 << 0), // => 0x00000001
46 
47  /**
48  * Should the model be recompiled?
49  * The LLVM ModelGenerator maintins a hash table of currently running
50  * models. If this flag is NOT set, then the generator will look to see
51  * if there is already a running instance of the given model and
52  * use the generated code from that one.
53  *
54  * If only a single instance of a model is run, there is no
55  * need to cache the models, and this can safetly be enabled,
56  * realizing some performance gains.
57  */
58  RECOMPILE = (0x1 << 1), // => 0x00000010
59 
60  /**
61  * If this is set, then a read-only model is generated. A read-only
62  * model can be simulated, but no code is generated to set model
63  * values, i.e. parameters, amounts, values, etc...
64  *
65  * It takes a finite amount of time to generate the model value setting
66  * functions, and if they are not needed, one may see some performance
67  * gains, especially in very large models.
68  */
69  READ_ONLY = (0x1 << 2), // => 0x00000100
70 
71  /**
72  * Generate accessor functions to allow changing of initial
73  * conditions.
74  */
75  MUTABLE_INITIAL_CONDITIONS = (0x1 << 3), // => 0x00001000
76 
77  /**
78  * GVN - This pass performs global value numbering and redundant load
79  * elimination cotemporaneously.
80  */
81  OPTIMIZE_GVN = (0x1 << 4),
82 
83  /**
84  * CFGSimplification - Merge basic blocks, eliminate unreachable blocks,
85  * simplify terminator instructions, etc...
86  */
87  OPTIMIZE_CFG_SIMPLIFICATION = (0x1 << 5),
88 
89  /**
90  * InstructionCombining - Combine instructions to form fewer, simple
91  * instructions. This pass does not modify the CFG, and has a tendency to make
92  * instructions dead, so a subsequent DCE pass is useful.
93  */
94  OPTIMIZE_INSTRUCTION_COMBINING = (0x1 << 6),
95 
96  /**
97  * DeadInstElimination - This pass quickly removes trivially dead instructions
98  * without modifying the CFG of the function. It is a BasicBlockPass, so it
99  * runs efficiently when queued next to other BasicBlockPass's.
100  */
101  OPTIMIZE_DEAD_INST_ELIMINATION = (0x1 << 7),
102 
103  /**
104  * DeadCodeElimination - This pass is more powerful than DeadInstElimination,
105  * because it is worklist driven that can potentially revisit instructions when
106  * their other instructions become dead, to eliminate chains of dead
107  * computations.
108  */
109  OPTIMIZE_DEAD_CODE_ELIMINATION = (0x1 << 8),
110 
111  /**
112  * InstructionSimplifier - Remove redundant instructions.
113  */
114  OPTIMIZE_INSTRUCTION_SIMPLIFIER = (0x1 << 9),
115 
116  /**
117  * all optimizations, use to check if bit mask has
118  * any optimizations.
119  */
120  OPTIMIZE = OPTIMIZE_GVN | OPTIMIZE_CFG_SIMPLIFICATION |
121  OPTIMIZE_INSTRUCTION_COMBINING |
122  OPTIMIZE_DEAD_INST_ELIMINATION | OPTIMIZE_DEAD_CODE_ELIMINATION |
123  OPTIMIZE_INSTRUCTION_SIMPLIFIER,
124 
125  /**
126  * Use the LLVM MCJIT JIT engine.
127  *
128  * Defaults to false.
129  *
130  * The MCJIT is the new LLVM JIT engine, it is not as well tested as the
131  * original JIT engine. Does NOT work on LLVM 3.1
132  */
133  USE_MCJIT = (0x1 << 10),
134 
135 
136  LLVM_SYMBOL_CACHE = (0x1 << 11)
137  };
