147  /**
148  * Simulate the current SBML model.
149  *
150  * If options is null, then the current simulation settings (start time,
151  * end time, n steps) are used. If options is not null, then the
152  * current simulation settings are set to the values specified by
153  * options and they are used.
154  *
155  * The options Dictionary may contain a large number of options, for a complete
156  * list of all available options for each integrator type, @see IntegratorFactory,
157  * @see SimulateOptions.
158  *
159  * For example, to perform a simulation from time 0 to 10 with 1000 steps, using a
160  * stiff integtator, you would:
161  * @code
162  * RoadRunner r = RoadRunner("someFile.xml");
163  * BasicDictionary opt;
164  * opt.setItem("start", 0);
165  * opt.setItem("duration", 10);
166  * opt.setItem("steps", 1000);
167  * opt.setItem("stiff", true);
168  * const DoubleMatrix *result = r.simulate(&opt);
169  * @endcode
170  *
171  * Similarly, if one wants to use a stochastic integrator, such as the Gillespie
172  * integrator, this is set via the "integrator" key, i.e.
173  * @code
174  * RoadRunner r = RoadRunner("someFile.xml");
175  * BasicDictionary opt;
176  * opt.setItem("integrator", "gillespie");
177  * opt.setItem("start", 0);
178  * opt.setItem("duration", 10);
179  * opt.setItem("steps", 1000);
180  * opt.setItem("stiff", true);
181  * opt.setItem("seed", 12345);
182  * const DoubleMatrix *result = r.simulate(&opt);
183  * @endcode
184  * Here, the "integrator" specifies the integrator to use. The "stiff" key
185  * is only used by the deterministic solvers, and it is safely ignored by the
186  * stochastic solvers. Also, the "seed" sets the random seed that the integrator
187  * uses. For more information about all of the avaialble options for each integrator,
188  * @see IntegratorFactory::getIntegratorOptions".
189  *
190  * @throws an std::exception if any options are invalid.
191  * @returns a borrowed reference to a DoubleMatrix object if successfull.
192  */
193  const ls::DoubleMatrix *simulate(const Dictionary* options = 0);
