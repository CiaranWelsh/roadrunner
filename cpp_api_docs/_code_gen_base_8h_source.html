95  template <size_t N>
96  llvm::BasicBlock *codeGenHeader(const char* functionName,
97  llvm::Type *retType,
98  llvm::Type* (&argTypes)[N],
99  const char* (&argNames)[N],
100  llvm::Value* (&args)[N])
101  {
102  // make the set init value function
103 
104  llvm::FunctionType *funcType = llvm::FunctionType::get(retType,
105  argTypes, false);
106  function = llvm::Function::Create(funcType,
107  llvm::Function::InternalLinkage,
108  functionName, module);
109 
110  // Create a new basic block to start insertion into.
111  llvm::BasicBlock *basicBlock = llvm::BasicBlock::Create(context,
112  "entry", function);
113  builder.SetInsertPoint(basicBlock);
114 
115  assert(function->getArgumentList().size() == N);
116 
117  int i = 0;
118  for (llvm::Function::arg_iterator ai = function->arg_begin();
119  ai != function->arg_end(); ++ai, ++i)
120  {
121  llvm::Value *arg = ai;
122  arg->setName(argNames[i]);
123  args[i] = arg;
124  }
125 
126  return basicBlock;
127  }

130  * the most common type of generated function takes a ModelData*,

132  */
133  llvm::BasicBlock *codeGenVoidModelDataHeader(const char* functionName,
134  llvm::Value* &modelData)
135  {
136  llvm::Type *argTypes[] = {

139  };
140 
141  const char *argNames[] = { "modelData" };

143  llvm::Value *args[] = { 0 };
144 
145  llvm::BasicBlock *basicBlock = codeGenHeader(functionName,
146  llvm::Type::getVoidTy(context),
147  argTypes, argNames, args);
148 
149  modelData = args[0];
150  return basicBlock;
151  }
