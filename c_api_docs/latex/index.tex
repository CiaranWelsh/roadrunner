    \hyperlink{rrc__types_8h_a1d68f0592372208fa5a5f2799ea4b3ae}{RRHandle} rrHandle;

    printf (\textcolor{stringliteral}{"Starting Test Program %s\(\backslash\)n"}, argv[0]);
